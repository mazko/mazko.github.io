/*
  xc8 --chip=16f648A pwm.c
*/

#include <xc.h>

#pragma config WDTE = OFF

// ignore two least significant bits PWMxDCL 
#define _DUTY_CYCLE(value) \
  CCPR1L = (4*(249+1)*value)>>2

int main() {

  // http://microchip.wikidot.com/8bit:10bitpwm

  // 0b01 Prescaler is 4; 0b1xx TMR2 on
  T2CON = 0b00000101;

  // Desired PWM frequency 250Hz
  // PR2=[(Fosc)/(4‚àóTMR2Prescale‚àóPWMFrequency)]‚àí1
  // PR2=(1Mhz/(4*4*250Hz))-1=249
  PR2 = 249;

  // 0b11xx - PWM mode;
  // assert log(1e6/249)/log(2) > 8 bit

  CCP1CON = 0b00001100;

  PORTB = TRISB = 0;

  while(1) {
    if (!RA1) {
      _DUTY_CYCLE(1/4);
    } else if (!RA2) {
      _DUTY_CYCLE(1/3);
    } else if (!RA3) {
      _DUTY_CYCLE(1/2);
    } else if (!RA4) {
      _DUTY_CYCLE(2/3);
    }
    // Do something useful
    for (unsigned char i=0; --i;);
  }

  return 0;
}
