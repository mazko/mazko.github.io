#include <xc.h>
#include <stdbool.h>

#pragma config WDTE = OFF

int main() {

  PORTB = TRISB = 0;

  while(true) {
    if (!RA1) {
      RB0 = true;
    } else if (!RA2) {
      RB0 = RB1 = true;
    } else if (!RA3) {
      RB0 = RB1 = RB2 = true;
    } else if (!RA4) {

    } else {
      PORTB = 0;
    }
  }

  return 0;
}
